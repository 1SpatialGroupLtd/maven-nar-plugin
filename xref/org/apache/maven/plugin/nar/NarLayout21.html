33  /**
34   * Layout which expands a nar file into:
35   * 
36   * <pre>
37   * nar/noarch/include
38   * nar/aol/<aol>-<type>/bin
39   * nar/aol/<aol>-<type>/lib
40   * </pre>
41   * 
42   * This loayout has a one-to-one relation with the aol-type version of the nar.
43   * 
44   * @author Mark Donszelmann (Mark.Donszelmann@gmail.com)
45   */
46  public class NarLayout21
47      extends AbstractNarLayout
48  {
49      private NarFileLayout fileLayout;
50  
51      public NarLayout21( Log log )
52      {
53          super( log );
54          this.fileLayout = new NarFileLayout10();
55      }

57      public File getNoArchDirectory( File baseDir, String artifactId, String version )
58      {
59          return new File( baseDir, artifactId + "-" + version + "-" + NarConstants.NAR_NO_ARCH );

81      public final File getLibDirectory( File baseDir, String artifactId, String version, String aol, String type )
82          throws MojoExecutionException
83      {
84          if ( type.equals( Library.EXECUTABLE ) )
85          {
86              throw new MojoExecutionException(
87                                                "NAR: for type EXECUTABLE call getBinDirectory instead of getLibDirectory" );
88          }
89  
90          File dir = getAolDirectory( baseDir, artifactId, version, aol, type );
91          dir = new File( dir, fileLayout.getLibDirectory( aol, type ) );
92          return dir;
93      }

107     /*
108      * (non-Javadoc)
109      * @see org.apache.maven.plugin.nar.NarLayout#attachNars(java.io.File, org.apache.maven.project.MavenProjectHelper,
110      * org.apache.maven.project.MavenProject, org.apache.maven.plugin.nar.NarInfo)
111      */
112     public final void attachNars( File baseDir, ArchiverManager archiverManager, MavenProjectHelper projectHelper,
113                                   MavenProject project, NarInfo narInfo )
114         throws MojoExecutionException
115     {
116         if ( getNoArchDirectory( baseDir, project.getArtifactId(), project.getVersion() ).exists() )
117         {
118             attachNar( archiverManager, projectHelper, project, NarConstants.NAR_NO_ARCH,
119                        getNoArchDirectory( baseDir, project.getArtifactId(), project.getVersion() ), "*/**" );
120             narInfo.setNar( null, NarConstants.NAR_NO_ARCH, project.getGroupId() + ":" + project.getArtifactId() + ":"
121                 + NarConstants.NAR_TYPE + ":" + NarConstants.NAR_NO_ARCH );
122         }
123 
124         // list all directories in basedir, scan them for classifiers
125         String[] subDirs = baseDir.list();
126         for ( int i = 0; ( subDirs != null ) && ( i < subDirs.length ); i++ )
127         {
128             String artifactIdVersion = project.getArtifactId() + "-" + project.getVersion();
129 
130             // skip entries not belonging to this project
131             if ( !subDirs[i].startsWith( artifactIdVersion ) )
132                 continue;
133 
134             String classifier = subDirs[i].substring( artifactIdVersion.length() + 1 );
135 
136             // skip noarch here
137             if ( classifier.equals( NarConstants.NAR_NO_ARCH ) )
138                 continue;
139 
140             File dir = new File( baseDir, subDirs[i] );
141             attachNar( archiverManager, projectHelper, project, classifier, dir, "*/**" );
142 
143             int lastDash = classifier.lastIndexOf( '-' );
144             String type = classifier.substring( lastDash + 1 );
145             AOL aol = new AOL( classifier.substring( 0, lastDash - 1 ) );
146 
147             if ( type.equals( Library.EXECUTABLE ) )
148             {
149                 if ( narInfo.getBinding( aol, null ) == null )
150                 {
151                     narInfo.setBinding( aol, Library.EXECUTABLE );
152                 }
153                 if ( narInfo.getBinding( null, null ) == null )
154                 {
155                     narInfo.setBinding( null, Library.EXECUTABLE );
156                 }
157             }
158             else
159             {
160                 // and not set or override if SHARED
161                 if ( ( narInfo.getBinding( aol, null ) == null ) || type.equals( Library.SHARED ) )
162                 {
163                     narInfo.setBinding( aol, type );
164                 }
165                 // and not set or override if SHARED
166                 if ( ( narInfo.getBinding( null, null ) == null ) || type.equals( Library.SHARED ) )
167                 {
168                     narInfo.setBinding( null, type );
169                 }
170             }
171 
172             narInfo.setNar( null, type, project.getGroupId() + ":" + project.getArtifactId() + ":"
173                 + NarConstants.NAR_TYPE + ":" + "${aol}" + "-" + type );
174         }
175     }

177     public void unpackNar( File unpackDirectory, ArchiverManager archiverManager, File file, String os, String linkerName,
178                            AOL defaultAOL )
179         throws MojoExecutionException, MojoFailureException
180     {
181         File dir = new File( unpackDirectory, FileUtils.basename( file.getPath(), "." + NarConstants.NAR_EXTENSION ) );
182 
183         boolean process = false;
184 
185         if ( !unpackDirectory.exists() )
186         {
187             unpackDirectory.mkdirs();
188             process = true;
189         }
190         else if ( !dir.exists() )
191         {
192             process = true;
193         }
194         else if ( file.lastModified() > dir.lastModified() )
195         {
196             try
197             {
198                 FileUtils.deleteDirectory( dir );
199             }
200             catch ( IOException e )
201             {
202                 throw new MojoExecutionException( "Could not delete directory: " + dir, e );
203             }
204 
205             process = true;
206         }
207 
208         if ( process )
209         {
210             unpackNarAndProcess( archiverManager, file, dir, os, linkerName, defaultAOL );
211         }
212     }
