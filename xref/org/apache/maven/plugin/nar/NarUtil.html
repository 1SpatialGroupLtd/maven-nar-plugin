73      public static String getOS( String defaultOs )
74      {
75          String os = defaultOs;
76          // adjust OS if not given
77          if ( os == null )
78          {
79              os = System.getProperty( "os.name" );
80              String name = os.toLowerCase();
81              if ( name.startsWith( "windows" ) )
82              {
83                  os = OS.WINDOWS;
84              }
85              if ( name.startsWith( "linux" ) )
86              {
87                  os = OS.LINUX;
88              }
89              if ( name.equals( "mac os x" ) )
90              {
91                  os = OS.MACOSX;
92              }
93          }
94          return os;
95      }

157     public static void makeExecutable( File file, final Log log )
158         throws MojoExecutionException, MojoFailureException
159     {
160         if ( !file.exists() )
161         {
162             return;
163         }
164 
165         if ( file.isDirectory() )
166         {
167             File[] files = file.listFiles();
168             for ( int i = 0; i < files.length; i++ )
169             {
170                 makeExecutable( files[i], log );
171             }
172         }
173         if ( file.isFile() && file.canRead() && file.canWrite() && !file.isHidden() )
174         {
175             // chmod +x file
176             int result = runCommand( "chmod", new String[] { "+x", file.getPath() }, null, null, log );
177             if ( result != 0 )
178             {
179                 throw new MojoExecutionException( "Failed to execute 'chmod +x " + file.getPath() + "'"
180                     + " return code: \'" + result + "\'." );
181             }
182         }
183     }

224             for ( int i = 0; i < files.length; i++ )
225             {
226                 makeLink( files[i], log );
227             }
228         }
229         if ( file.isFile() && file.canRead() && file.canWrite() && !file.isHidden()
230             && file.getName().matches( ".*\\.so(\\.\\d+)+$" ) )
231         {
232             File sofile =
233                 new File( file.getParent(), file.getName().substring( 0, file.getName().indexOf( ".so" ) + 3 ) );
234             if ( !sofile.exists() )
235             {
236                 // ln -s lib.so.xx lib.so
237                 int result =
238                     runCommand( "ln", new String[] { "-s", file.getName(), sofile.getPath() }, null, null, log );
239                 if ( result != 0 )
240                 {
241                     throw new MojoExecutionException( "Failed to execute 'ln -s " + file.getName() + " "
242                         + sofile.getPath() + "'" + " return code: \'" + result + "\'." );
243                 }
244             }
245         }

263      * Returns the header file name (javah) corresponding to the given class file name
264      * 
265      * @param filename the absolute file name of the class
266      * @return the header file name.
267      */
268     public static String getHeaderName( String basename, String filename )
269     {
270         String base = basename.replaceAll( "\\\\", "/" );
271         String file = filename.replaceAll( "\\\\", "/" );
272         if ( !file.startsWith( base ) )
273         {
274             throw new IllegalArgumentException( "Error " + file + " does not start with " + base );
275         }
276         String header = file.substring( base.length() + 1 );
277         header = header.replaceAll( "/", "_" );
278         header = header.replaceAll( "\\.class", ".h" );
279         return header;
280     }

298     /* for jdk 1.4 */
299     private static String quote( String s )
300     {
301         final String escQ = "\\Q";
302         final String escE = "\\E";
303 
304         int slashEIndex = s.indexOf( escE );
305         if ( slashEIndex == -1 )
306         {
307             return escQ + s + escE;
308         }
309 
310         StringBuffer sb = new StringBuffer( s.length() * 2 );
311         sb.append( escQ );
312         slashEIndex = 0;
313         int current = 0;
314         while ( ( slashEIndex = s.indexOf( escE, current ) ) != -1 )
315         {
316             sb.append( s.substring( current, slashEIndex ) );
317             current = slashEIndex + 2;
318             sb.append( escE );
319             sb.append( "\\" );
320             sb.append( escE );
321             sb.append( escQ );
322         }
323         sb.append( s.substring( current, s.length() ) );
324         sb.append( escE );
325         return sb.toString();
326     }

328     /* for jdk 1.4 */
329     private static String quoteReplacement( String s )
330     {
331         if ( ( s.indexOf( '\\' ) == -1 ) && ( s.indexOf( '$' ) == -1 ) )
332         {
333             return s;
334         }
335         StringBuffer sb = new StringBuffer();
336         for ( int i = 0; i < s.length(); i++ )
337         {
338             char c = s.charAt( i );
339             if ( c == '\\' )
340             {
341                 sb.append( '\\' );
342                 sb.append( '\\' );
343             }
344             else if ( c == '$' )
345             {
346                 sb.append( '\\' );
347                 sb.append( '$' );
348             }
349             else
350             {
351                 sb.append( c );
352             }
353         }
354         return sb.toString();
355     }

361     public static int copyDirectoryStructure( File sourceDirectory, File destinationDirectory, String includes,
362                                               String excludes )
363         throws IOException
364     {
365         if ( !sourceDirectory.exists() )
366         {
367             throw new IOException( "Source directory doesn't exists (" + sourceDirectory.getAbsolutePath() + ")." );
368         }
369 
370         List files = FileUtils.getFiles( sourceDirectory, includes, excludes );
371         String sourcePath = sourceDirectory.getAbsolutePath();
372 
373         int copied = 0;
374         for ( Iterator i = files.iterator(); i.hasNext(); )
375         {
376             File file = (File) i.next();
377             String dest = file.getAbsolutePath();
378             dest = dest.substring( sourcePath.length() + 1 );
379             File destination = new File( destinationDirectory, dest );
380             if ( file.isFile() )
381             {
382                 // destination = destination.getParentFile();
383                 FileUtils.copyFile( file, destination );
384                 copied++;
385 
386                 // copy executable bit
387                 try
388                 {
389                     // 1.6 only so coded using introspection
390                     // destination.setExecutable( file.canExecute(), false );
391                     Method canExecute = file.getClass().getDeclaredMethod( "canExecute", new Class[] {} );
392                     Method setExecutable =
393                         destination.getClass().getDeclaredMethod( "setExecutable",
394                                                                   new Class[] { boolean.class, boolean.class } );
395                     setExecutable.invoke( destination, new Object[] {
396                         (Boolean) canExecute.invoke( file, new Object[] {} ), Boolean.FALSE } );
397                 }
398                 catch ( SecurityException e )
399                 {
400                     // ignored
401                 }
402                 catch ( NoSuchMethodException e )
403                 {
404                     // ignored
405                 }
406                 catch ( IllegalArgumentException e )
407                 {
408                     // ignored
409                 }
410                 catch ( IllegalAccessException e )
411                 {
412                     // ignored
413                 }
414                 catch ( InvocationTargetException e )
415                 {
416                     // ignored
417                 }
418             }
419             else if ( file.isDirectory() )
420             {
421                 if ( !destination.exists() && !destination.mkdirs() )
422                 {
423                     throw new IOException( "Could not create destination directory '" + destination.getAbsolutePath()
424                         + "'." );
425                 }
426                 copied += copyDirectoryStructure( file, destination, includes, excludes );
427             }
428             else
429             {
430                 throw new IOException( "Unknown file type: " + file.getAbsolutePath() );
431             }
432         }
433         return copied;
434     }

464     public static String addLibraryPathToEnv( String path, Map environment, String os )
465     {
466         String pathName = null;
467         char separator = ' ';
468         if ( os.equals( OS.WINDOWS ) )
469         {
470             pathName = "Path";
471             separator = ';';
472         }
473         else if ( os.equals( OS.MACOSX ) )
474         {
475             pathName = "DYLD_LIBRARY_PATH";
476             separator = ':';
477         }
478         else
479         {
480             pathName = "LD_LIBRARY_PATH";
481             separator = ':';
482         }
483 
484         String value = environment != null ? (String) environment.get( pathName ) : null;
485         if ( value == null )
486         {
487             value = NarUtil.getEnv( pathName, pathName, null );
488         }
489 
490         String libPath = path;
491         libPath = libPath.replace( File.pathSeparatorChar, separator );
492         if ( value != null )
493         {
494             value += separator + libPath;
495         }
496         else
497         {
498             value = libPath;
499         }
500         if ( environment != null )
501         {
502             environment.put( pathName, value );
503         }
504         return pathName + "=" + value;
505     }

532     public static int runCommand( String cmd, String[] args, File workingDirectory, String[] env, TextStream out,
533                                   TextStream err, TextStream dbg )
534         throws MojoExecutionException, MojoFailureException
535     {
536         Commandline cmdLine = new Commandline();
537 
538         try
539         {
540             dbg.println( "RunCommand: " + cmd );
541             cmdLine.setExecutable( cmd );
542             if ( args != null )
543             {
544                 for ( int i = 0; i < args.length; i++ )
545                 {
546                     dbg.println( "  '" + args[i] + "'" );
547                 }
548                 cmdLine.addArguments( args );
549             }
550             if ( workingDirectory != null )
551             {
552                 dbg.println( "in: " + workingDirectory.getPath() );
553                 cmdLine.setWorkingDirectory( workingDirectory );
554             }
555 
556             if ( env != null )
557             {
558                 dbg.println( "with Env:" );
559                 for ( int i = 0; i < env.length; i++ )
560                 {
561                     String[] nameValue = env[i].split( "=", 2 );
562                     if ( nameValue.length < 2 )
563                     {
564                         throw new MojoFailureException( "   Misformed env: '" + env[i] + "'" );
565                     }
566                     dbg.println( "   '" + nameValue[0] + "=" + nameValue[1] + "'" );
567                     cmdLine.addEnvironment( nameValue[0], nameValue[1] );
568                 }
569             }
570 
571             Process process = cmdLine.execute();
572             StreamGobbler errorGobbler = new StreamGobbler( process.getErrorStream(), err );
573             StreamGobbler outputGobbler = new StreamGobbler( process.getInputStream(), out );
574 
575             errorGobbler.start();
576             outputGobbler.start();
577             process.waitFor();
578             dbg.println( "ExitValue: " + process.exitValue() );
579             final int timeout = 5000;
580             errorGobbler.join( timeout );
581             outputGobbler.join( timeout );
582             return process.exitValue();
583         }
584         catch ( Exception e )
585         {
586             throw new MojoExecutionException( "Could not launch " + cmdLine, e );
587         }
588     }

590     private static final class StreamGobbler
591         extends Thread
592     {
593         private InputStream is;
594 
595         private TextStream ts;
596 
597         private StreamGobbler( InputStream is, TextStream ts )
598         {
599             this.is = is;
600             this.ts = ts;
601         }
602 
603         public void run()
604         {
605             try
606             {
607                 BufferedReader reader = new BufferedReader( new InputStreamReader( is ) );
608                 String line = null;
609                 while ( ( line = reader.readLine() ) != null )
610                 {
611                     ts.println( line );
612                 }
613                 reader.close();
614             }
615             catch ( IOException e )
616             {
617                 // e.printStackTrace()
618                 StackTraceElement[] stackTrace = e.getStackTrace();
619                 for ( int i = 0; i < stackTrace.length; i++ )
620                 {
621                     ts.println( stackTrace[i].toString() );
622                 }
623             }
624         }
625     }
