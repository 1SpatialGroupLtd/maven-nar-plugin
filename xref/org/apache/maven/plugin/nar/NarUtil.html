66      {

68          if ( os == null )
69          {
70              os = System.getProperty( "os.name" );
71              String name = os.toLowerCase();
72              if ( name.startsWith( "windows" ) )
73                  os = OS.WINDOWS;

75                  os = OS.LINUX;

140     public static void makeExecutable( File file, final Log log )
141         throws MojoExecutionException, MojoFailureException
142     {
143         if ( !file.exists() )
144             return;
145 
146         if ( file.isDirectory() )
147         {
148             File[] files = file.listFiles();
149             for ( int i = 0; i < files.length; i++ )
150             {
151                 makeExecutable( files[i], log );
152             }
153         }
154         if ( file.isFile() && file.canRead() && file.canWrite() && !file.isHidden() )
155         {
156             // chmod +x file
157             int result = runCommand( "chmod", new String[] { "+x", file.getPath() }, null, null, log );
158             if ( result != 0 )
159             {
160                 throw new MojoExecutionException( "Failed to execute 'chmod +x " + file.getPath() + "'"
161                     + " return code: \'" + result + "\'." );
162             }
163         }
164     }

209      * Returns the header file name (javah) corresponding to the given class file name
210      * 
211      * @param filename the absolute file name of the class
212      * @return the header file name.
213      */
214     public static final String getHeaderName( String base, String filename )
215     {
216         base = base.replaceAll( "\\\\", "/" );
217         filename = filename.replaceAll( "\\\\", "/" );
218         if ( !filename.startsWith( base ) )
219         {
220             throw new IllegalArgumentException( "Error " + filename + " does not start with " + base );
221         }
222         String header = filename.substring( base.length() + 1 );
223         header = header.replaceAll( "/", "_" );
224         header = header.replaceAll( "\\.class", ".h" );
225         return header;
226     }

244     /* for jdk 1.4 */
245     private static String quote( String s )
246     {
247         int slashEIndex = s.indexOf( "\\E" );
248         if ( slashEIndex == -1 )
249             return "\\Q" + s + "\\E";
250 
251         StringBuffer sb = new StringBuffer( s.length() * 2 );
252         sb.append( "\\Q" );
253         slashEIndex = 0;
254         int current = 0;
255         while ( ( slashEIndex = s.indexOf( "\\E", current ) ) != -1 )
256         {
257             sb.append( s.substring( current, slashEIndex ) );
258             current = slashEIndex + 2;
259             sb.append( "\\E\\\\E\\Q" );
260         }
261         sb.append( s.substring( current, s.length() ) );
262         sb.append( "\\E" );
263         return sb.toString();
264     }

266     /* for jdk 1.4 */
267     private static String quoteReplacement( String s )
268     {
269         if ( ( s.indexOf( '\\' ) == -1 ) && ( s.indexOf( '$' ) == -1 ) )
270             return s;
271         StringBuffer sb = new StringBuffer();
272         for ( int i = 0; i < s.length(); i++ )
273         {
274             char c = s.charAt( i );
275             if ( c == '\\' )
276             {
277                 sb.append( '\\' );
278                 sb.append( '\\' );
279             }
280             else if ( c == '$' )
281             {
282                 sb.append( '\\' );
283                 sb.append( '$' );
284             }
285             else
286             {
287                 sb.append( c );
288             }
289         }
290         return sb.toString();
291     }

297     public static int copyDirectoryStructure( File sourceDirectory, File destinationDirectory, String includes,
298                                               String excludes )
299         throws IOException
300     {
301         if ( !sourceDirectory.exists() )
302         {
303             throw new IOException( "Source directory doesn't exists (" + sourceDirectory.getAbsolutePath() + ")." );
304         }
305 
306         List files = FileUtils.getFiles( sourceDirectory, includes, excludes );
307         String sourcePath = sourceDirectory.getAbsolutePath();
308 
309         int copied = 0;
310         for ( Iterator i = files.iterator(); i.hasNext(); )
311         {
312             File file = (File) i.next();
313             String dest = file.getAbsolutePath();
314             dest = dest.substring( sourcePath.length() + 1 );
315             File destination = new File( destinationDirectory, dest );
316             if ( file.isFile() )
317             {
318                 destination = destination.getParentFile();
319                 FileUtils.copyFileToDirectory( file, destination );
320                 copied++;
321             }
322             else if ( file.isDirectory() )
323             {
324                 if ( !destination.exists() && !destination.mkdirs() )
325                 {
326                     throw new IOException( "Could not create destination directory '" + destination.getAbsolutePath()
327                         + "'." );
328                 }
329                 copied += copyDirectoryStructure( file, destination, includes, excludes );
330             }
331             else
332             {
333                 throw new IOException( "Unknown file type: " + file.getAbsolutePath() );
334             }
335         }
336         return copied;
337     }

367     public static String addLibraryPathToEnv( String path, Map environment, String os )
368     {
369         String pathName = null;
370         char separator = ' ';
371         if ( os.equals( OS.WINDOWS ) )
372         {
373             pathName = "Path";
374             separator = ';';
375         }
376         else if ( os.equals( OS.MACOSX ) )
377         {
378             pathName = "DYLD_LIBRARY_PATH";
379             separator = ':';
380         }
381         else
382         {
383             pathName = "LD_LIBRARY_PATH";
384             separator = ':';
385         }
386 
387         String value = environment != null ? (String) environment.get( pathName ) : null;
388         if ( value == null )
389         {
390             value = NarUtil.getEnv( pathName, pathName, null );
391         }
392 
393         path = path.replace( File.pathSeparatorChar, separator );
394         if ( value != null )
395         {
396             value += separator + path;
397         }
398         else
399         {
400             value = path;
401         }
402         if ( environment != null )
403         {
404             environment.put( pathName, value );
405         }
406         return pathName + "=" + value;
407     }

409     public static int runCommand( String cmd, String[] args, File workingDirectory, String[] env, Log log )
410         throws MojoExecutionException, MojoFailureException
411     {
412         Commandline cmdLine = new Commandline();
413         
414         try
415         {
416             log.debug( "RunCommand: " + cmd );
417             cmdLine.setExecutable( cmd );
418             if ( args != null )
419             {
420                 for ( int i = 0; i < args.length; i++ )
421                 {
422                     log.debug( "  '" + args[i] + "'" );
423                 }
424                 cmdLine.addArguments( args );
425             }
426             if ( workingDirectory != null )
427             {
428                 log.debug( "in: " + workingDirectory.getPath() );
429                 cmdLine.setWorkingDirectory( workingDirectory );
430             }
431 
432             if ( env != null )
433             {
434                 log.debug( "with Env:" );
435                 for ( int i = 0; i < env.length; i++ )
436                 {
437                     String[] nameValue = env[i].split( "=", 2 );
438                     if ( nameValue.length < 2 )
439                         throw new MojoFailureException( "   Misformed env: '" + env[i] + "'" );
440                     log.debug( "   '" + nameValue[0] + "=" + nameValue[1] + "'" );
441                     cmdLine.addEnvironment( nameValue[0], nameValue[1] );
442                 }
443             }
444 
445             Process process = cmdLine.execute();
446             StreamGobbler errorGobbler = new StreamGobbler( process.getErrorStream(), true, log );
447             StreamGobbler outputGobbler = new StreamGobbler( process.getInputStream(), false, log );
448 
449             errorGobbler.start();
450             outputGobbler.start();
451             process.waitFor();
452             return process.exitValue();
453         }
454         catch ( Throwable e )
455         {
456             throw new MojoExecutionException( "Could not launch " + cmdLine, e );
457         }
458     }

461     static class StreamGobbler
462         extends Thread
463     {
464         InputStream is;
465 
466         boolean error;
467 
468         Log log;
469 
470         StreamGobbler( InputStream is, boolean error, Log log )
471         {
472             this.is = is;
473             this.error = error;
474             this.log = log;
475         }
476 
477         public void run()
478         {
479             try
480             {
481                 BufferedReader reader = new BufferedReader( new InputStreamReader( is ) );
482                 String line = null;
483                 while ( ( line = reader.readLine() ) != null )
484                 {
485                     if ( error )
486                     {
487                         log.error( line );
488                     }
489                     else
490                     {
491                         log.debug( line );
492                     }
493                 }
494                 reader.close();
495             }
496             catch ( IOException e )
497             {
498                 e.printStackTrace();
499             }
500         }
501     }
