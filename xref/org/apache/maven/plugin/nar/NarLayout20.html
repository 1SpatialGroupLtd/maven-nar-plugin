33  /**
34   * Initial layout which expands a nar file into:
35   * 
36   * <pre>
37   * nar/includue
38   * nar/bin
39   * nar/lib
40   * </pre>
41   * 
42   * this layout was abandoned because there is no one-to-one relation between the nar file and its directory structure.
43   * Therefore SNAPSHOTS could not be fully deleted when replaced.
44   * 
45   * @author Mark Donszelmann (Mark.Donszelmann@gmail.com)
46   */

104      * (non-Javadoc)
105      * @see org.apache.maven.plugin.nar.NarLayout#attachNars(java.io.File, org.apache.maven.project.MavenProjectHelper,
106      * org.apache.maven.project.MavenProject, org.apache.maven.plugin.nar.NarInfo)

108     public final void attachNars( File baseDir, ArchiverManager archiverManager, MavenProjectHelper projectHelper,
109                                   MavenProject project, NarInfo narInfo )
110         throws MojoExecutionException
111     {
112         if ( getIncludeDirectory( baseDir, project.getArtifactId(), project.getVersion() ).exists() )
113         {
114             attachNar( archiverManager, projectHelper, project, "noarch", baseDir, "include/**" );
115             narInfo.setNar( null, "noarch", project.getGroupId() + ":" + project.getArtifactId() + ":"
116                 + NarConstants.NAR_TYPE + ":" + "noarch" );
117         }
118 
119         String[] binAOL = new File( baseDir, "bin" ).list();
120         for ( int i = 0; ( binAOL != null ) && ( i < binAOL.length ); i++ )
121         {
122             attachNar( archiverManager, projectHelper, project, binAOL[i] + "-" + Library.EXECUTABLE, baseDir, "bin/"
123                 + binAOL[i] + "/**" );
124             narInfo.setNar( null, Library.EXECUTABLE, project.getGroupId() + ":" + project.getArtifactId() + ":"
125                 + NarConstants.NAR_TYPE + ":" + "${aol}" + "-" + Library.EXECUTABLE );
126             narInfo.setBinding( new AOL( binAOL[i] ), Library.EXECUTABLE );
127             narInfo.setBinding( null, Library.EXECUTABLE );
128         }
129 
130         File libDir = new File( baseDir, "lib" );
131         String[] libAOL = libDir.list();
132         for ( int i = 0; ( libAOL != null ) && ( i < libAOL.length ); i++ )
133         {
134             String bindingType = null;
135             String[] libType = new File( libDir, libAOL[i] ).list();
136             for ( int j = 0; ( libType != null ) && ( j < libType.length ); j++ )
137             {
138                 attachNar( archiverManager, projectHelper, project, libAOL[i] + "-" + libType[j], baseDir, "lib/"
139                     + libAOL[i] + "/" + libType[j] + "/**" );
140                 narInfo.setNar( null, libType[j], project.getGroupId() + ":" + project.getArtifactId() + ":"
141                     + NarConstants.NAR_TYPE + ":" + "${aol}" + "-" + libType[j] );
142 
143                 // set if not set or override if SHARED
144                 if ( ( bindingType == null ) || libType[j].equals( Library.SHARED ) )
145                 {
146                     bindingType = libType[j];
147                 }
148             }
149 
150             AOL aol = new AOL( libAOL[i] );
151             if ( narInfo.getBinding( aol, null ) == null )
152             {
153                 narInfo.setBinding( aol, bindingType != null ? bindingType : Library.NONE );
154             }
155             if ( narInfo.getBinding( null, null ) == null )
156             {
157                 narInfo.setBinding( null, bindingType != null ? bindingType : Library.NONE );
158             }
159         }
160     }

162     public void unpackNar( File unpackDir, ArchiverManager archiverManager, File file, String os, String linkerName,
163                            AOL defaultAOL )
164         throws MojoExecutionException, MojoFailureException
165     {
166         File flagFile =
167             new File( unpackDir, FileUtils.basename( file.getPath(), "." + NarConstants.NAR_EXTENSION ) + ".flag" );
168 
169         boolean process = false;
170         if ( !unpackDir.exists() )
171         {
172             unpackDir.mkdirs();
173             process = true;
174         }
175         else if ( !flagFile.exists() )
176         {
177             process = true;
178         }
179         else if ( file.lastModified() > flagFile.lastModified() )
180         {
181             process = true;
182         }
183 
184         if ( process )
185         {
186             try
187             {
188                 unpackNarAndProcess( archiverManager, file, unpackDir, os, linkerName, defaultAOL );
189                 FileUtils.fileDelete( flagFile.getPath() );
190                 FileUtils.fileWrite( flagFile.getPath(), "" );
191             }
192             catch ( IOException e )
193             {
194                 throw new MojoFailureException( "Cannot create flag file: " + flagFile.getPath(), e );
195             }
196         }
197     }
