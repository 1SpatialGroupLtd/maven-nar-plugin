57      protected final void attachNar( ArchiverManager archiverManager, MavenProjectHelper projectHelper,
58                                      MavenProject project, String classifier, File dir, String include )
59          throws MojoExecutionException
60      {
61          File narFile =
62              new File( project.getBuild().getDirectory(), project.getBuild().getFinalName() + "-" + classifier + "."
63                  + NarConstants.NAR_EXTENSION );
64          if ( narFile.exists() )
65          {
66              narFile.delete();
67          }
68          try
69          {
70              Archiver archiver = archiverManager.getArchiver( NarConstants.NAR_ROLE_HINT );
71              archiver.addDirectory( dir, new String[] { include }, null );
72              archiver.setDestFile( narFile );
73              archiver.createArchive();
74          }
75          catch ( NoSuchArchiverException e )
76          {
77              throw new MojoExecutionException( "NAR: cannot find archiver", e );
78          }
79          catch ( ArchiverException e )
80          {
81              throw new MojoExecutionException( "NAR: cannot create NAR archive '" + narFile + "'", e );
82          }
83          catch ( IOException e )
84          {
85              throw new MojoExecutionException( "NAR: cannot create NAR archive '" + narFile + "'", e );
86          }
87          projectHelper.attachArtifact( project, NarConstants.NAR_TYPE, classifier, narFile );
88      }

90      protected void unpackNarAndProcess( ArchiverManager archiverManager, File file, File narLocation, String os,
91                                          String linkerName, AOL defaultAOL )
92          throws MojoExecutionException, MojoFailureException
93      {
94  
95          final String gpp = "g++";
96          final String gcc = "gcc";
97  
98          narLocation.mkdirs();
99  
100         // unpack
101         try
102         {
103             UnArchiver unArchiver;
104             unArchiver = archiverManager.getUnArchiver( NarConstants.NAR_ROLE_HINT );
105             unArchiver.setSourceFile( file );
106             unArchiver.setDestDirectory( narLocation );
107             unArchiver.extract();
108         }
109         catch ( NoSuchArchiverException e )
110         {
111             throw new MojoExecutionException( "Error unpacking file: " + file + " to: " + narLocation, e );
112         }
113         catch ( ArchiverException e )
114         {
115             throw new MojoExecutionException( "Error unpacking file: " + file + " to: " + narLocation, e );
116         }
117 
118         // process
119         if ( !NarUtil.getOS( os ).equals( OS.WINDOWS ) )
120         {
121             NarUtil.makeExecutable( new File( narLocation, "bin/" + defaultAOL ), log );
122             // FIXME clumsy
123             if ( defaultAOL.hasLinker( gpp ) )
124             {
125                 NarUtil.makeExecutable( new File( narLocation, "bin/"
126                     + NarUtil.replace( gpp, gcc, defaultAOL.toString() ) ), log );
127             }
128             // add link to versioned so files
129             NarUtil.makeLink( new File( narLocation, "lib/" + defaultAOL ), log );
130         }
131         if ( linkerName.equals( gcc ) || linkerName.equals( gpp ) )
132         {
133             NarUtil.runRanlib( new File( narLocation, "lib/" + defaultAOL ), log );
134             // FIXME clumsy
135             if ( defaultAOL.hasLinker( gpp ) )
136             {
137                 NarUtil.runRanlib(
138                                    new File( narLocation, "lib/" + NarUtil.replace( gpp, gcc, defaultAOL.toString() ) ),
139                                    log );
140             }
141         }
142     }

144     /**
145      * @return
146      * @throws MojoExecutionException
147      */
148     public static NarLayout getLayout( String layoutName, Log log )
149         throws MojoExecutionException
150     {
151         String className =
152             layoutName.indexOf( '.' ) < 0 ? NarLayout21.class.getPackage().getName() + "." + layoutName : layoutName;
153         log.debug( "Using " + className );
154         Class cls;
155         try
156         {
157             cls = Class.forName( className );
158             Constructor ctor = cls.getConstructor( new Class[] { Log.class } );
159             return (NarLayout) ctor.newInstance( new Object[] { log } );
160         }
161         catch ( ClassNotFoundException e )
162         {
163             throw new MojoExecutionException( "Cannot find class for layout " + className, e );
164         }
165         catch ( InstantiationException e )
166         {
167             throw new MojoExecutionException( "Cannot instantiate class for layout " + className, e );
168         }
169         catch ( IllegalAccessException e )
170         {
171             throw new MojoExecutionException( "Cannot access class for layout " + className, e );
172         }
173         catch ( SecurityException e )
174         {
175             throw new MojoExecutionException( "Cannot access class for layout " + className, e );
176         }
177         catch ( NoSuchMethodException e )
178         {
179             throw new MojoExecutionException( "Cannot find ctor(Log) for layout " + className, e );
180         }
181         catch ( IllegalArgumentException e )
182         {
183             throw new MojoExecutionException( "Wrong arguments ctor(Log) for layout " + className, e );
184         }
185         catch ( InvocationTargetException e )
186         {
187             throw new MojoExecutionException( "Cannot invokector(Log) for layout " + className, e );
188         }
189     }
