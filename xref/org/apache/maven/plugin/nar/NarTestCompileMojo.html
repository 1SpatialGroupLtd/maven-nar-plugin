42  /**
43   * Compiles native test source files.
44   * 
45   * @goal nar-testCompile
46   * @phase test-compile
47   * @requiresDependencyResolution test
48   * @author Mark Donszelmann
49   */
50  public class NarTestCompileMojo
51      extends AbstractCompileMojo
52  {
53      /**
54       * Skip running of NAR integration test plugins.
55       * 
56       * @parameter expression="${skipNar}" default-value="false"
57       */
58      private boolean skipNar;
59  
60      public final void narExecute()
61          throws MojoExecutionException, MojoFailureException
62      {
63          // make sure destination is there
64          getTestTargetDirectory().mkdirs();
65  
66          for ( Iterator i = getTests().iterator(); i.hasNext(); )
67          {
68              createTest( getAntProject(), (Test) i.next() );
69          }
70      }
71  
72      private void createTest( Project antProject, Test test )
73          throws MojoExecutionException, MojoFailureException
74      {
75          String type = "test";
76  
77          // configure task
78          CCTask task = new CCTask();
79          task.setProject( antProject );
80  
81          // subsystem
82          SubsystemEnum subSystem = new SubsystemEnum();
83          subSystem.setValue( "console" );
84          task.setSubsystem( subSystem );
85  
86          // outtype
87          OutputTypeEnum outTypeEnum = new OutputTypeEnum();
88          outTypeEnum.setValue( Library.EXECUTABLE );
89          task.setOuttype( outTypeEnum );
90  
91          // outDir
92          File outDir = new File( getTestTargetDirectory(), "bin" );
93          outDir = new File( outDir, getAOL().toString() );
94          outDir.mkdirs();
95  
96          // outFile
97          File outFile = new File( outDir, test.getName() );
98          getLog().debug( "NAR - output: '" + outFile + "'" );
99          task.setOutfile( outFile );
100 
101         // object directory
102         File objDir = new File( getTestTargetDirectory(), "obj" );
103         objDir = new File( objDir, getAOL().toString() );
104         objDir.mkdirs();
105         task.setObjdir( objDir );
106 
107         // failOnError, libtool
108         task.setFailonerror( failOnError( getAOL() ) );
109         task.setLibtool( useLibtool( getAOL() ) );
110 
111         // runtime
112         RuntimeType runtimeType = new RuntimeType();
113         runtimeType.setValue( getRuntime( getAOL() ) );
114         task.setRuntime( runtimeType );
115 
116         // add C++ compiler
117         CompilerDef cpp = getCpp().getCompiler( type, test.getName() );
118         if ( cpp != null )
119         {
120             task.addConfiguredCompiler( cpp );
121         }
122 
123         // add C compiler
124         CompilerDef c = getC().getCompiler( type, test.getName() );
125         if ( c != null )
126         {
127             task.addConfiguredCompiler( c );
128         }
129 
130         // add Fortran compiler
131         CompilerDef fortran = getFortran().getCompiler( type, test.getName() );
132         if ( fortran != null )
133         {
134             task.addConfiguredCompiler( fortran );
135         }
136 
137         // add java include paths
138         getJava().addIncludePaths( task, type );
139 
140         // add dependency include paths
141         for ( Iterator i = getNarManager().getNarDependencies( "test" ).iterator(); i.hasNext(); )
142         {
143             Artifact artifact = (Artifact) i.next();
144             File include =
145                 getLayout().getIncludeDirectory( getUnpackDirectory(), artifact.getArtifactId(), artifact.getVersion() );
146             if ( include.exists() )
147             {
148                 task.createIncludePath().setPath( include.getPath() );
149             }
150         }
151 
152         // add linker
153         task.addConfiguredLinker( getLinker().getLinker( this, antProject, getOS(), getAOL().getKey() + "linker.", type ) );
154 
155         // FIXME hardcoded values
156         String libName = getFinalName();
157         File includeDir =
158             getLayout().getIncludeDirectory( getTargetDirectory(), getMavenProject().getArtifactId(),
159                                              getMavenProject().getVersion() );
160         File libDir =
161             getLayout().getLibDirectory( getTargetDirectory(), getMavenProject().getArtifactId(),
162                                          getMavenProject().getVersion(), getAOL().toString(), test.getLink() );
163 
164         // copy shared library
165         // FIXME why do we do this ?
166         /*
167          * Removed in alpha-10 if (test.getLink().equals(Library.SHARED)) { try { // defaults are Unix String libPrefix
168          * = NarUtil.getDefaults().getProperty( getAOLKey() + "shared.prefix", "lib"); String libExt =
169          * NarUtil.getDefaults().getProperty( getAOLKey() + "shared.extension", "so"); File copyDir = new
170          * File(getTargetDirectory(), (getOS().equals( "Windows") ? "bin" : "lib") + "/" + getAOL() + "/" +
171          * test.getLink()); FileUtils.copyFileToDirectory(new File(libDir, libPrefix + libName + "." + libExt),
172          * copyDir); if (!getOS().equals(OS.WINDOWS)) { libDir = copyDir; } } catch (IOException e) { throw new
173          * MojoExecutionException( "NAR: Could not copy shared library", e); } }
174          */
175         // FIXME what about copying the other shared libs?
176 
177         // add include of this package
178         if ( includeDir.exists() )
179         {
180             task.createIncludePath().setLocation( includeDir );
181         }
182 
183         // add library of this package
184         if ( libDir.exists() )
185         {
186             LibrarySet libSet = new LibrarySet();
187             libSet.setProject( antProject );
188             libSet.setLibs( new CUtil.StringArrayBuilder( libName ) );
189             LibraryTypeEnum libType = new LibraryTypeEnum();
190             libType.setValue( test.getLink() );
191             libSet.setType( libType );
192             libSet.setDir( libDir );
193             task.addLibset( libSet );
194         }
195 
196         // add dependency libraries
197         List depLibOrder = getDependencyLibOrder();
198         List depLibs = getNarManager().getNarDependencies( "test" );
199 
200         // reorder the libraries that come from the nar dependencies
201         // to comply with the order specified by the user
202         if ( ( depLibOrder != null ) && !depLibOrder.isEmpty() )
203         {
204 
205             List tmp = new LinkedList();
206 
207             for ( Iterator i = depLibOrder.iterator(); i.hasNext(); )
208             {
209 
210                 String depToOrderName = (String) i.next();
211 
212                 for ( Iterator j = depLibs.iterator(); j.hasNext(); )
213                 {
214 
215                     NarArtifact dep = (NarArtifact) j.next();
216                     String depName = dep.getGroupId() + ":" + dep.getArtifactId();
217 
218                     if ( depName.equals( depToOrderName ) )
219                     {
220 
221                         tmp.add( dep );
222                         j.remove();
223                     }
224                 }
225             }
226 
227             tmp.addAll( depLibs );
228             depLibs = tmp;
229         }
230 
231         for ( Iterator i = depLibs.iterator(); i.hasNext(); )
232         {
233 
234             Artifact dependency = (Artifact) i.next();
235             // FIXME: this should be preferred binding
236             File libDirectory =
237                 getLayout().getLibDirectory( getUnpackDirectory(), dependency.getArtifactId(), dependency.getVersion(),
238                                              getAOL().toString(), test.getLink() );
239             if ( libDirectory.exists() )
240             {
241                 LibrarySet libset = new LibrarySet();
242                 libset.setProject( antProject );
243                 libset.setLibs( new CUtil.StringArrayBuilder( dependency.getArtifactId() + "-"
244                     + dependency.getVersion() ) );
245                 libset.setDir( libDirectory );
246                 task.addLibset( libset );
247             }
248         }
249 
250         // Add JVM to linker
251         getJava().addRuntime( task, getJavaHome( getAOL() ), getOS(), getAOL().getKey() + ".java." );
252 
253         // execute
254         try
255         {
256             task.execute();
257         }
258         catch ( BuildException e )
259         {
260             throw new MojoExecutionException( "NAR: Test-Compile failed", e );
261         }
262     }
263 
264     protected final File getTestTargetDirectory()
265     {
266         return new File( getMavenProject().getBuild().getDirectory(), "test-nar" );
267     }
