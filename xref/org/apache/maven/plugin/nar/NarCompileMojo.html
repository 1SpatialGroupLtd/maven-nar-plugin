51  /**
52   * Compiles native source files.
53   * 
54   * @goal nar-compile
55   * @phase compile
56   * @requiresSession
57   * @requiresProject
58   * @requiresDependencyResolution compile
59   * @author Mark Donszelmann
60   */
61  public class NarCompileMojo
62      extends AbstractCompileMojo
63  {
64      /**
65       * The current build session instance.
66       * 
67       * @parameter expression="${session}"
68       * @required
69       * @readonly
70       */
71      private MavenSession session;

73      public final void narExecute()
74          throws MojoExecutionException, MojoFailureException
75      {
76          for ( Iterator i = session.getSortedProjects().iterator(); i.hasNext(); )
77          {
78              MavenProject project = (MavenProject) i.next();
79              if ( !project.getPackaging().equals( NarConstants.NAR ) )
80              {
81                  continue;
82              }
83              if ( project.isExecutionRoot() )
84              {
85                  continue;
86              }
87  
88              // is this me ? bail out, the list was sorted
89              MavenProject me = getMavenProject();
90              if ( project.getArtifact().equals( me.getArtifact() ) )
91              {
92                  break;
93              }
94  
95              // search the dependency list
96              for ( Iterator it = me.getArtifacts().iterator(); it.hasNext(); )
97              {
98                  Artifact dependency = (Artifact) it.next();
99                  // equals will not work here as the project type is "nar" while the dependencies type is "jar"
100                 // 
101                 if ( dependency.getArtifactId().equals( project.getArtifactId() )
102                     && dependency.getGroupId().equals( project.getGroupId() )
103                     && dependency.getVersion().equals( project.getVersion() ) && dependency.getType().equals( "jar" ) )
104                 {
105                     getLog().info(
106                                    "Added intermodule dependency to " + project.getArtifact() + " in "
107                                        + project.getBuild().getDirectory() );
108                 }
109             }
110         }
111 
112         // make sure destination is there
113         getTargetDirectory().mkdirs();
114 
115         // check for source files
116         int noOfSources = 0;
117         noOfSources += getSourcesFor( getCpp() ).size();
118         noOfSources += getSourcesFor( getC() ).size();
119         noOfSources += getSourcesFor( getFortran() ).size();
120         if ( noOfSources > 0 )
121         {
122             getLog().info( "Compiling " + noOfSources + " native files" );
123             for ( Iterator i = getLibraries().iterator(); i.hasNext(); )
124             {
125                 createLibrary( getAntProject(), (Library) i.next() );
126             }
127         }
128         else
129         {
130             getLog().info( "Nothing to compile" );
131         }
132 
133         try
134         {
135             // FIXME, should the include paths be defined at a higher level ?
136             getCpp().copyIncludeFiles(
137                                        getMavenProject(),
138                                        getLayout().getIncludeDirectory( getTargetDirectory(),
139                                                                         getMavenProject().getArtifactId(),
140                                                                         getMavenProject().getVersion() ) );
141         }
142         catch ( IOException e )
143         {
144             throw new MojoExecutionException( "NAR: could not copy include files", e );
145         }
146     }

148     private List getSourcesFor( Compiler compiler )
149         throws MojoFailureException, MojoExecutionException
150     {
151         try
152         {
153             List files = new ArrayList();
154             List srcDirs = compiler.getSourceDirectories();
155             for ( Iterator i = srcDirs.iterator(); i.hasNext(); )
156             {
157                 File dir = (File) i.next();
158                 if ( dir.exists() )
159                 {
160                     files.addAll( FileUtils.getFiles( dir, StringUtils.join( compiler.getIncludes().iterator(), "," ),
161                                                       null ) );
162                 }
163             }
164             return files;
165         }
166         catch ( IOException e )
167         {
168             return Collections.EMPTY_LIST;
169         }
170     }

172     private void createLibrary( Project antProject, Library library )
173         throws MojoExecutionException, MojoFailureException
174     {
175         getLog().debug( "Creating Library " + library );
176         // configure task
177         CCTask task = new CCTask();
178         task.setProject( antProject );
179 
180         // subsystem
181         SubsystemEnum subSystem = new SubsystemEnum();
182         subSystem.setValue( library.getSubSystem() );
183         task.setSubsystem( subSystem );
184 
185         // set max cores
186         task.setMaxCores( getMaxCores( getAOL() ) );
187 
188         // outtype
189         OutputTypeEnum outTypeEnum = new OutputTypeEnum();
190         String type = library.getType();
191         outTypeEnum.setValue( type );
192         task.setOuttype( outTypeEnum );
193 
194         // stdc++
195         task.setLinkCPP( library.linkCPP() );
196 
197         // fortran
198         task.setLinkFortran( library.linkFortran() );
199         task.setLinkFortranMain( library.linkFortranMain() );
200 
201         // outDir
202         File outDir;
203         if ( type.equals( Library.EXECUTABLE ) )
204         {
205             outDir =
206                 getLayout().getBinDirectory( getTargetDirectory(), getMavenProject().getArtifactId(),
207                                              getMavenProject().getVersion(), getAOL().toString() );
208         }
209         else
210         {
211             outDir =
212                 getLayout().getLibDirectory( getTargetDirectory(), getMavenProject().getArtifactId(),
213                                              getMavenProject().getVersion(), getAOL().toString(), type );
214         }
215         outDir.mkdirs();
216 
217         // outFile
218         // FIXME NAR-90 we could get the final name from layout
219         File outFile;
220         if ( type.equals( Library.EXECUTABLE ) )
221         {
222             // executable has no version number
223             outFile = new File( outDir, getMavenProject().getArtifactId() );
224         }
225         else
226         {
227             outFile = new File( outDir, getOutput( getAOL() ) );
228         }
229         getLog().debug( "NAR - output: '" + outFile + "'" );
230         task.setOutfile( outFile );
231 
232         // object directory
233         File objDir = new File( getTargetDirectory(), "obj" );
234         objDir = new File( objDir, getAOL().toString() );
235         objDir.mkdirs();
236         task.setObjdir( objDir );
237 
238         // failOnError, libtool
239         task.setFailonerror( failOnError( getAOL() ) );
240         task.setLibtool( useLibtool( getAOL() ) );
241 
242         // runtime
243         RuntimeType runtimeType = new RuntimeType();
244         runtimeType.setValue( getRuntime( getAOL() ) );
245         task.setRuntime( runtimeType );
246 
247         // add C++ compiler
248         CompilerDef cpp = getCpp().getCompiler( type, getOutput( getAOL() ) );
249         if ( cpp != null )
250         {
251             task.addConfiguredCompiler( cpp );
252         }
253 
254         // add C compiler
255         CompilerDef c = getC().getCompiler( type, getOutput( getAOL() ) );
256         if ( c != null )
257         {
258             task.addConfiguredCompiler( c );
259         }
260 
261         // add Fortran compiler
262         CompilerDef fortran = getFortran().getCompiler( type, getOutput( getAOL() ) );
263         if ( fortran != null )
264         {
265             task.addConfiguredCompiler( fortran );
266         }
267 
268         // add javah include path
269         File jniDirectory = getJavah().getJniDirectory();
270         if ( jniDirectory.exists() )
271         {
272             task.createIncludePath().setPath( jniDirectory.getPath() );
273         }
274 
275         // add java include paths
276         getJava().addIncludePaths( task, type );
277 
278         // add dependency include paths
279         for ( Iterator i = getNarManager().getNarDependencies( "compile" ).iterator(); i.hasNext(); )
280         {
281             // FIXME, handle multiple includes from one NAR
282             NarArtifact narDependency = (NarArtifact) i.next();
283             String binding = narDependency.getNarInfo().getBinding( getAOL(), Library.STATIC );
284             getLog().debug( "Looking for " + narDependency + " found binding " + binding );
285             if ( !binding.equals( Library.JNI ) )
286             {
287                 File unpackDirectory = getUnpackDirectory();
288                 File include =
289                     getLayout().getIncludeDirectory( unpackDirectory, narDependency.getArtifactId(),
290                                                      narDependency.getVersion() );
291                 getLog().debug( "Looking for include directory: " + include );
292                 if ( include.exists() )
293                 {
294                     task.createIncludePath().setPath( include.getPath() );
295                 }
296             }
297         }
298 
299         // add linker
300         LinkerDef linkerDefinition =
301             getLinker().getLinker( this, antProject, getOS(), getAOL().getKey() + ".linker.", type );
302         task.addConfiguredLinker( linkerDefinition );
303 
304         // add dependency libraries
305         // FIXME: what about PLUGIN and STATIC, depending on STATIC, should we
306         // not add all libraries, see NARPLUGIN-96
307         if ( type.equals( Library.SHARED ) || type.equals( Library.JNI ) || type.equals( Library.EXECUTABLE ) )
308         {
309 
310             List depLibOrder = getDependencyLibOrder();
311             List depLibs = getNarManager().getNarDependencies( "compile" );
312 
313             // reorder the libraries that come from the nar dependencies
314             // to comply with the order specified by the user
315             if ( ( depLibOrder != null ) && !depLibOrder.isEmpty() )
316             {
317                 List tmp = new LinkedList();
318 
319                 for ( Iterator i = depLibOrder.iterator(); i.hasNext(); )
320                 {
321                     String depToOrderName = (String) i.next();
322 
323                     for ( Iterator j = depLibs.iterator(); j.hasNext(); )
324                     {
325                         NarArtifact dep = (NarArtifact) j.next();
326                         String depName = dep.getGroupId() + ":" + dep.getArtifactId();
327 
328                         if ( depName.equals( depToOrderName ) )
329                         {
330                             tmp.add( dep );
331                             j.remove();
332                         }
333                     }
334                 }
335 
336                 tmp.addAll( depLibs );
337                 depLibs = tmp;
338             }
339 
340             for ( Iterator i = depLibs.iterator(); i.hasNext(); )
341             {
342                 NarArtifact dependency = (NarArtifact) i.next();
343 
344                 // FIXME no handling of "local"
345 
346                 // FIXME, no way to override this at this stage
347                 String binding = dependency.getNarInfo().getBinding( getAOL(), Library.STATIC );
348                 getLog().debug( "Using Binding: " + binding );
349                 AOL aol = getAOL();
350                 aol = dependency.getNarInfo().getAOL( getAOL() );
351                 getLog().debug( "Using Library AOL: " + aol.toString() );
352 
353                 if ( !binding.equals( Library.JNI ) )
354                 {
355                     File unpackDirectory = getUnpackDirectory();
356 
357                     File dir =
358                         getLayout().getLibDirectory( unpackDirectory, dependency.getArtifactId(),
359                                                      dependency.getVersion(), aol.toString(), binding );
360 
361                     getLog().debug( "Looking for Library Directory: " + dir );
362                     if ( dir.exists() )
363                     {
364                         LibrarySet libSet = new LibrarySet();
365                         libSet.setProject( antProject );
366 
367                         // FIXME, no way to override
368                         String libs = dependency.getNarInfo().getLibs( getAOL() );
369                         if ( ( libs != null ) && !libs.equals( "" ) )
370                         {
371                             getLog().debug( "Using LIBS = " + libs );
372                             libSet.setLibs( new CUtil.StringArrayBuilder( libs ) );
373                             libSet.setDir( dir );
374                             task.addLibset( libSet );
375                         }
376                     }
377                     else
378                     {
379                         getLog().debug( "Library Directory " + dir + " does NOT exist." );
380                     }
381 
382                     // FIXME, look again at this, for multiple dependencies we may need to remove duplicates
383                     String options = dependency.getNarInfo().getOptions( getAOL() );
384                     if ( ( options != null ) && !options.equals( "" ) )
385                     {
386                         getLog().debug( "Using OPTIONS = " + options );
387                         LinkerArgument arg = new LinkerArgument();
388                         arg.setValue( options );
389                         linkerDefinition.addConfiguredLinkerArg( arg );
390                     }
391 
392                     String sysLibs = dependency.getNarInfo().getSysLibs( getAOL() );
393                     if ( ( sysLibs != null ) && !sysLibs.equals( "" ) )
394                     {
395                         getLog().debug( "Using SYSLIBS = " + sysLibs );
396                         SystemLibrarySet sysLibSet = new SystemLibrarySet();
397                         sysLibSet.setProject( antProject );
398 
399                         sysLibSet.setLibs( new CUtil.StringArrayBuilder( sysLibs ) );
400                         task.addSyslibset( sysLibSet );
401                     }
402                 }
403             }
404         }
405 
406         // Add JVM to linker
407         getJava().addRuntime( task, getJavaHome( getAOL() ), getOS(), getAOL().getKey() + ".java." );
408 
409         // execute
410         try
411         {
412             task.execute();
413         }
414         catch ( BuildException e )
415         {
416             throw new MojoExecutionException( "NAR: Compile failed", e );
417         }
418 
419         // FIXME, this should be done in CPPTasks at some point
420         if ( getRuntime( getAOL() ).equals( "dynamic" ) && getOS().equals( OS.WINDOWS )
421             && getLinker().getName( null, null ).equals( "msvc" ) && !getLinker().getVersion().startsWith( "6." ) )
422         {
423             String libType = library.getType();
424             if ( libType.equals( Library.JNI ) || libType.equals( Library.SHARED ) )
425             {
426                 String dll = outFile.getPath() + ".dll";
427                 String manifest = dll + ".manifest";
428                 int result =
429                     NarUtil.runCommand( "mt.exe", new String[] { "/manifest", manifest,
430                         "/outputresource:" + dll + ";#2" }, null, null, getLog() );
431                 if ( result != 0 )
432                 {
433                     throw new MojoFailureException( "MT.EXE failed with exit code: " + result );
434                 }
435             }
436         }
437     }
