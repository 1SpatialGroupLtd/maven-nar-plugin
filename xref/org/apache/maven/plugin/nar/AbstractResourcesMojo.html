22  import java.io.File;
23  import java.io.IOException;
24  import java.util.Iterator;
25  import java.util.List;

27  import org.apache.maven.plugin.MojoExecutionException;
28  import org.apache.maven.plugin.MojoFailureException;
29  import org.codehaus.plexus.archiver.ArchiverException;
30  import org.codehaus.plexus.archiver.UnArchiver;
31  import org.codehaus.plexus.archiver.manager.ArchiverManager;
32  import org.codehaus.plexus.archiver.manager.NoSuchArchiverException;
33  import org.codehaus.plexus.util.FileUtils;

35  /**
36   * Keeps track of resources
37   * 
38   * @author Mark Donszelmann
39   */
40  public abstract class AbstractResourcesMojo
41      extends AbstractCompileMojo
42  {
43      /**
44       * Binary directory
45       * 
46       * @parameter expression="bin"
47       * @required
48       */
49      private String resourceBinDir;
50  
51      /**
52       * Include directory
53       * 
54       * @parameter expression="include"
55       * @required
56       */
57      private String resourceIncludeDir;
58  
59      /**
60       * Library directory
61       * 
62       * @parameter expression="lib"
63       * @required
64       */
65      private String resourceLibDir;
66  
67      /**
68       * To look up Archiver/UnArchiver implementations
69       * 
70       * @component role="org.codehaus.plexus.archiver.manager.ArchiverManager"
71       * @required
72       */
73      private ArchiverManager archiverManager;
74  
75      protected final int copyIncludes( File srcDir )
76          throws IOException, MojoExecutionException, MojoFailureException
77      {
78          int copied = 0;
79  
80          // copy includes
81          File includeDir = new File( srcDir, resourceIncludeDir );
82          if ( includeDir.exists() )
83          {
84              File includeDstDir = getLayout().getIncludeDirectory( getTargetDirectory() );
85              getLog().debug( "Copying includes from " + includeDir + " to " + includeDstDir );
86              copied += NarUtil.copyDirectoryStructure( includeDir, includeDstDir, null, NarUtil.DEFAULT_EXCLUDES );
87          }
88  
89          return copied;
90      }
91  
92      protected final int copyBinaries( File srcDir, String aol )
93          throws IOException, MojoExecutionException, MojoFailureException
94      {
95          int copied = 0;
96  
97          // copy binaries
98          File binDir = new File( srcDir, resourceBinDir );
99          if ( binDir.exists() )
100         {
101             File binDstDir = getLayout().getBinDirectory( getTargetDirectory(), aol );
102             getLog().debug( "Copying binaries from " + binDir + " to " + binDstDir );
103             copied += NarUtil.copyDirectoryStructure( binDir, binDstDir, null, NarUtil.DEFAULT_EXCLUDES );
104         }
105 
106         return copied;
107     }
108 
109     protected final int copyLibraries( File srcDir, String aol )
110         throws MojoFailureException, IOException, MojoExecutionException
111     {
112         int copied = 0;
113 
114         // copy libraries
115         File libDir = new File( srcDir, resourceLibDir );
116         if ( libDir.exists() )
117         {
118             // create all types of libs
119             for ( Iterator i = getLibraries().iterator(); i.hasNext(); )
120             {
121                 Library library = (Library) i.next();
122                 String type = library.getType();
123                 File libDstDir = getLayout().getLibDirectory( getTargetDirectory(), aol, type );
124                 getLog().debug( "Copying libraries from " + libDir + " to " + libDstDir );
125 
126                 // filter files for lib
127                 String includes =
128                     "**/*." + NarUtil.getDefaults().getProperty( NarUtil.getAOLKey( aol ) + "." + type + ".extension" );
129 
130                 // add import lib for Windows shared libraries
131                 if ( new AOL( aol ).getOS().equals( OS.WINDOWS ) && type.equals( Library.SHARED ) )
132                 {
133                     includes += ",**/*.lib";
134                 }
135                 copied += NarUtil.copyDirectoryStructure( libDir, libDstDir, includes, NarUtil.DEFAULT_EXCLUDES );
136             }
137         }
138 
139         return copied;
140     }
141 
142     protected final void copyResources( File srcDir, String aol )
143         throws MojoExecutionException, MojoFailureException
144     {
145         int copied = 0;
146         try
147         {
148             copied += copyIncludes( srcDir );
149 
150             copied += copyBinaries( srcDir, aol );
151 
152             copied += copyLibraries( srcDir, aol );
153 
154             // unpack jar files
155             File classesDirectory = new File( getOutputDirectory(), "classes" );
156             classesDirectory.mkdirs();
157             List jars = FileUtils.getFiles( srcDir, "**/*.jar", null );
158             for ( Iterator i = jars.iterator(); i.hasNext(); )
159             {
160                 File jar = (File) i.next();
161                 getLog().debug( "Unpacking jar " + jar );
162                 UnArchiver unArchiver;
163                 unArchiver = archiverManager.getUnArchiver( NarConstants.NAR_ROLE_HINT );
164                 unArchiver.setSourceFile( jar );
165                 unArchiver.setDestDirectory( classesDirectory );
166                 unArchiver.extract();
167             }
168         }
169         catch ( IOException e )
170         {
171             throw new MojoExecutionException( "NAR: Could not copy resources", e );
172         }
173         catch ( NoSuchArchiverException e )
174         {
175             throw new MojoExecutionException( "NAR: Could not find archiver", e );
176         }
177         catch ( ArchiverException e )
178         {
179             throw new MojoExecutionException( "NAR: Could not unarchive jar file", e );
180         }
181         getLog().info( "Copied " + copied + " resources for " + aol );
182     }
