35  /**
36   * Tests NAR files. Runs Native Tests and executables if produced.
37   * 
38   * @goal nar-test
39   * @phase test
40   * @requiresProject
41   * @author Mark Donszelmann
42   */
43  public class NarTestMojo
44      extends AbstractCompileMojo
45  {
46      /**
47       * The classpath elements of the project being tested.
48       * 
49       * @parameter expression="${project.testClasspathElements}"
50       * @required
51       * @readonly
52       */
53      private List classpathElements;
54  
55      public final void narExecute()
56          throws MojoExecutionException, MojoFailureException
57      {
58          // run all tests
59          for ( Iterator i = getTests().iterator(); i.hasNext(); )
60          {
61              runTest( (Test) i.next() );
62          }
63  
64          for ( Iterator i = getLibraries().iterator(); i.hasNext(); )
65          {
66              runExecutable( (Library) i.next() );
67          }
68      }
69  
70      private void runTest( Test test )
71          throws MojoExecutionException, MojoFailureException
72      {
73          // run if requested
74          if ( test.shouldRun() )
75          {
76              String name = getTestTargetDirectory().getPath() + "/bin/" + getAOL() + "/" + test.getName();
77              getLog().info( "Running test " + name );
78  
79              File workingDir = getMavenProject().getBasedir();
80              getLog().info( "  in " + workingDir );
81              List args = test.getArgs();
82              int result =
83                  NarUtil.runCommand( name, (String[]) args.toArray( new String[args.size()] ), workingDir,
84                                      generateEnvironment(), getLog() );
85              if ( result != 0 )
86              {
87                  throw new MojoFailureException( "Test " + name + " failed with exit code: " + result + " 0x"
88                      + Integer.toHexString( result ) );
89              }
90          }
91      }
92  
93      private void runExecutable( Library library )
94          throws MojoExecutionException, MojoFailureException
95      {
96          if ( library.getType().equals( Library.EXECUTABLE ) && library.shouldRun() )
97          {
98              MavenProject project = getMavenProject();
99              // FIXME NAR-90, we could make sure we get the final name from layout
100             File executable =
101                 new File( getLayout().getBinDirectory( getTargetDirectory(), getMavenProject().getArtifactId(),
102                                                        getMavenProject().getVersion(), getAOL().toString() ),
103                           project.getArtifactId() );
104             getLog().info( "Running executable " + executable );
105             List args = library.getArgs();
106             int result =
107                 NarUtil.runCommand( executable.getPath(), (String[]) args.toArray( new String[args.size()] ), null,
108                                     generateEnvironment(), getLog() );
109             if ( result != 0 )
110             {
111                 throw new MojoFailureException( "Test " + executable + " failed with exit code: " + result + " 0x"
112                     + Integer.toHexString( result ) );
113             }
114         }
115     }
116 
117     protected final File getTestTargetDirectory()
118     {
119         return new File( getMavenProject().getBuild().getDirectory(), "test-nar" );
120     }
121 
122     private String[] generateEnvironment()
123         throws MojoExecutionException, MojoFailureException
124     {
125         List env = new ArrayList();
126 
127         Set/* <File> */sharedPaths = new HashSet();
128 
129         // add all shared libraries of this package
130         for ( Iterator i = getLibraries().iterator(); i.hasNext(); )
131         {
132             Library lib = (Library) i.next();
133             if ( lib.getType().equals( Library.SHARED ) )
134             {
135                 File path =
136                     getLayout().getLibDirectory( getTargetDirectory(), getMavenProject().getArtifactId(),
137                                                  getMavenProject().getVersion(), getAOL().toString(), lib.getType() );
138                 getLog().debug( "Adding path to shared library: " + path );
139                 sharedPaths.add( path );
140             }
141         }
142 
143         // add dependent shared libraries
144         String classifier = getAOL() + "-shared";
145         List narArtifacts = getNarManager().getNarDependencies( "compile" );
146         List dependencies = getNarManager().getAttachedNarDependencies( narArtifacts, classifier );
147         for ( Iterator d = dependencies.iterator(); d.hasNext(); )
148         {
149             Artifact dependency = (Artifact) d.next();
150             getLog().debug( "Looking for dependency " + dependency );
151 
152             // FIXME reported to maven developer list, isSnapshot
153             // changes behaviour
154             // of getBaseVersion, called in pathOf.
155             dependency.isSnapshot();
156 
157             File libDirectory =
158                 getLayout().getLibDirectory( getUnpackDirectory(), dependency.getArtifactId(), dependency.getVersion(),
159                                              getAOL().toString(), Library.SHARED );
160             sharedPaths.add( libDirectory );
161         }
162 
163         // set environment
164         if ( sharedPaths.size() > 0 )
165         {
166             String sharedPath = "";
167             for ( Iterator i = sharedPaths.iterator(); i.hasNext(); )
168             {
169                 sharedPath += ( (File) i.next() ).getPath();
170                 if ( i.hasNext() )
171                 {
172                     sharedPath += File.pathSeparator;
173                 }
174             }
175 
176             String sharedEnv = NarUtil.addLibraryPathToEnv( sharedPath, null, getOS() );
177             env.add( sharedEnv );
178         }
179 
180         // necessary to find WinSxS
181         if ( getOS().equals( OS.WINDOWS ) )
182         {
183             env.add( "SystemRoot=" + NarUtil.getEnv( "SystemRoot", "SystemRoot", "C:\\Windows" ) );
184         }
185 
186         // add CLASSPATH
187         env.add( "CLASSPATH=" + StringUtils.join( classpathElements.iterator(), File.pathSeparator ) );
188 
189         return env.size() > 0 ? (String[]) env.toArray( new String[env.size()] ) : null;
190     }
